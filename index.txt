    let myChartInstance = null;

    function app() {
      return {
        // App State
        currentPage: 'dash', activeTab: 'hist', search: '', searchPart: '',

        isLogDetailView: false, isEditingEquip: false, isEditingPart: false, isEditingLog: false,
        selectedEquip: null, notifications:[], isLoading: true,
        
        // Data State
        equipment: [], allParts:[], logs:[],

        logForm: { logId: '', type: 'PM', desc: '', downtime: 0, cost: 0, equipmentId: '', date: '', tech: 'Maintenance Team', parts:[], status: 'Completed', hm: '', catatan: '' },

        oldLogParts:[],

          { id: 'parts', name: 'Spare Parts', icon: 'fas fa-box', mobile: true },
          { id: 'hist', name: 'All Logs', icon: 'fas fa-history', mobile: true },
        ],

        // --- INIT & LISTENERS ---
        init() {
          this.setupFirebaseListeners();
          
          // Watchers for Chart
          this.$watch('equipment', () => { if(this.currentPage === 'dash' && !this.isLoading) this.renderChart(); });
          this.$watch('currentPage', (val) => { if(val === 'dash') setTimeout(() => this.renderChart(), 100); });
        },

        setupFirebaseListeners() {
          this.isLoading = true;

          // Listen to Equipment
          window.onValue(window.ref(window.db, 'Equipment'), (snapshot) => {
            const data = snapshot.val();
            this.equipment = data ? Object.values(data) :[];
            // Update selectedEquip if in detail view
            if(this.selectedEquip) {
              const updated = this.equipment.find(e => e.EquipmentID === this.selectedEquip.EquipmentID);
              if(updated) this.selectedEquip = updated;
            }
          });

          // Listen to Spare Parts
          window.onValue(window.ref(window.db, 'SpareParts'), (snapshot) => {
            const data = snapshot.val();
            this.allParts = data ? Object.values(data) :[];
          });

          // Listen to History Logs
          window.onValue(window.ref(window.db, 'HistoryLog'), (snapshot) => {
            const data = snapshot.val();
            // Sort by LogID descending (newest first)
            this.logs = data ? Object.values(data).sort((a, b) => String(b.LogID).localeCompare(String(a.LogID))) :[];
            this.isLoading = false;
          });
        },

        // --- HELPER FUNCTIONS ---
        formatDateForInput(dateStr) {
          if (!dateStr) return '';
          try {
            if (dateStr.includes('/')) {
              const parts = dateStr.split('/');
              if(parts.length === 3) return `${parts[2]}-${parts[1].padStart(2,'0')}-${parts[0].padStart(2,'0')}`;
            }
            const d = new Date(dateStr);
            if (!isNaN(d.getTime())) return d.toISOString().split('T')[0];
          } catch(e){}
          return '';
        },

        showNotification(message, type = 'success') {
          const id = Date.now();
          this.notifications.push({ id, message, type });
          setTimeout(() => { this.notifications = this.notifications.filter(n => n.id !== id); }, 4000);
        },

        // --- GETTERS (COMPUTED PROPS) ---
        get calculatedStats() {
          let overdueCount = 0;
          const today = new Date(); today.setHours(0,0,0,0);
          this.equipment.forEach(e => {
            if (e.NextPMDate) {
              let pmDate = new Date(e.NextPMDate);
              if (isNaN(pmDate.getTime()) && e.NextPMDate.includes('/')) {
                const parts = e.NextPMDate.split('/');
                if(parts.length === 3) pmDate = new Date(`${parts[2]}-${parts[1]}-${parts[0]}`);
              }
              if (pmDate < today) overdueCount++;
            }
          });
          const lowStockCount = this.allParts.filter(p => Number(p.Stok) <= Number(p.MinStock)).length;
          const totalDown = this.logs.reduce((acc, curr) => acc + Number(curr.Downtime || 0), 0);
          return {
            totalEquip: { label: 'Total Assets', value: this.equipment.length, color: 'border-blue-500' },
            overduePM: { label: 'Overdue PM', value: overdueCount, color: 'border-red-500' },
            lowStock: { label: 'Low Stock Parts', value: lowStockCount, color: 'border-yellow-500' },
            totalDowntime: { label: 'Total Down Hrs', value: parseFloat(totalDown).toFixed(1), color: 'border-green-500' }
          };
        },

        get filteredEquip() { 
          return (this.equipment ||[]).filter(e => 
            (e.Nama || '').toLowerCase().includes(this.search.toLowerCase()) || 
            (e.EquipmentID || '').toLowerCase().includes(this.search.toLowerCase())
          ); 
        },
        get filteredAllParts() { 
          return (this.allParts ||[]).filter(p => 
            (p.NamaPart || '').toLowerCase().includes(this.searchPart.toLowerCase()) || 
            (p.PartID || '').toLowerCase().includes(this.searchPart.toLowerCase())
          );

'bg-yellow-500/20 text-yellow-500 border border-yellow-500/30' : s === 'Decommissioned' ? 'bg-gray-500/20 text-gray-400 border border-gray-500/30' : 'bg-red-500/20 text-red-500 border border-red-500/30'; },

        },

        async submitLog() {
          this.isLoading = true;
          try {
            const logId = this.isEditingLog ? this.logForm.logId : "LOG-" + new Date().getTime();
            
            const logDataToSave = {
              LogID: logId, EquipmentID: this.logForm.equipmentId, Tanggal: this.logForm.date, 
              Jenis: this.logForm.type, Deskripsi: this.logForm.desc, Technician: this.logForm.tech,
              PartsUsed: JSON.stringify(this.logForm.parts), Downtime: Number(this.logForm.downtime),
              Cost: Number(this.logForm.cost), Status: this.logForm.status, HM: this.logForm.hm, Catatan: this.logForm.catatan
            };

            const updates = {};
            updates['/HistoryLog/' + logId] = logDataToSave;

            // --- STOCK ADJUSTMENT LOGIC (ATOMIC) ---
            if (!this.isEditingLog && this.logForm.parts.length > 0) {
              // DEDUCT NEW
              this.logForm.parts.forEach(p => {
                if(!p.id) return;
                const currentPart = this.allParts.find(x => x.PartID === p.id);
                if (currentPart) {
                  let currentStok = updates['/SpareParts/' + p.id + '/Stok'] !== undefined ? updates['/SpareParts/' + p.id + '/Stok'] : Number(currentPart.Stok);
                  updates['/SpareParts/' + p.id + '/Stok'] = currentStok - Number(p.qty);
                }
              });
            } else if (this.isEditingLog) {
              // ROLLBACK OLD + DEDUCT NEW
              // 1. Add back old parts
              this.oldLogParts.forEach(p => {
                if(!p.id) return;
                const currentPart = this.allParts.find(x => x.PartID === p.id);
                if (currentPart) {
                   let currentStok = updates['/SpareParts/' + p.id + '/Stok'] !== undefined ? updates['/SpareParts/' + p.id + '/Stok'] : Number(currentPart.Stok);
                   updates['/SpareParts/' + p.id + '/Stok'] = currentStok + Number(p.qty);
                }
              });
              // 2. Deduct new parts
              this.logForm.parts.forEach(p => {
                if(!p.id) return;
                const currentPart = this.allParts.find(x => x.PartID === p.id);
                if (currentPart) {
                  let currentStok = updates['/SpareParts/' + p.id + '/Stok'] !== undefined ? updates['/SpareParts/' + p.id + '/Stok'] : Number(currentPart.Stok);
                  updates['/SpareParts/' + p.id + '/Stok'] = currentStok - Number(p.qty);
                }
              });
            }

            // Execute atomic update
            await window.update(window.ref(window.db), updates);
            
            this.showNotification("Log saved successfully!");
            this.showLogModal = false;
          } catch(e) { 
            this.showNotification("Error: " + e.message, "error"); 
          } finally { this.isLoading = false; }
        },

        async deleteItem(nodeName, id) {
          if (!confirm(`Are you sure you want to delete this from ${nodeName}? This action cannot be undone.`)) return;
          this.isLoading = true;
          try {
            
            // If deleting a HistoryLog, rollback the stock first!
            if (nodeName === 'HistoryLog') {
               const logToDelete = this.logs.find(l => l.LogID === id);
               if (logToDelete && logToDelete.PartsUsed) {
                  let parts =[];
                  try { parts = typeof logToDelete.PartsUsed === 'string' ? JSON.parse(logToDelete.PartsUsed) : logToDelete.PartsUsed; } catch(e){}
                  
                  if (parts.length > 0) {
                     const updates = {};
                     updates[`/${nodeName}/${id}`] = null; // Mark log for deletion
                     
                     // Add back stock
                     parts.forEach(p => {
                        if(!p.id) return;
                        const currentPart = this.allParts.find(x => x.PartID === p.id);
                        if (currentPart) {
                           let currentStok = updates['/SpareParts/' + p.id + '/Stok'] !== undefined ? updates['/SpareParts/' + p.id + '/Stok'] : Number(currentPart.Stok);
                           updates['/SpareParts/' + p.id + '/Stok'] = currentStok + Number(p.qty);
                        }
                     });
                     
                     await window.update(window.ref(window.db), updates);
                     this.showNotification("Log deleted & stock returned!");
                     this.isLoading = false;
                     return;
                  }
               }
            }

            // Normal Delete
            await window.remove(window.ref(window.db, `${nodeName}/${id}`));
            this.showNotification("Item deleted successfully!");
            if (nodeName === 'Equipment' && this.currentPage === 'detail') this.currentPage = 'equip';
            
          } catch(e) { 
            this.showNotification("Delete Error: " + e.message, "error"); 
          } finally { this.isLoading = false; }
        },

        // --- CHART JS RENDERER ---
        renderChart() {
          const ctx = document.getElementById('statusChart');
          if(!ctx) return;
          
          const counts = { 'Active': 0, 'In Maintenance': 0, 'Idle': 0, 'Decommissioned': 0 };
          this.equipment.forEach(e => { 
            const st = e.Status ? e.Status.trim() : 'Idle'; 
            if(counts[st] !== undefined) counts[st]++; 
            else counts['Idle']++; // fallback
          });
          
          if (myChartInstance) myChartInstance.destroy();
          
          // Only render non-zero datasets for cleaner look
          const labels = [];
          const data =[];
          const bgColors = [];
          
          if(counts['Active'] > 0) { labels.push('Active'); data.push(counts['Active']); bgColors.push('#10B981'); }
          if(counts['In Maintenance'] > 0) { labels.push('In Maint.'); data.push(counts['In Maintenance']); bgColors.push('#F59E0B'); }
          if(counts['Idle'] > 0) { labels.push('Idle'); data.push(counts['Idle']); bgColors.push('#6B7280'); }
          if(counts['Decommissioned'] > 0) { labels.push('Decomm.'); data.push(counts['Decommissioned']); bgColors.push('#EF4444'); }

          if(data.length === 0) {
             labels.push('No Data'); data.push(1); bgColors.push('#374151');
          }

          myChartInstance = new Chart(ctx, {
            type: 'doughnut',
            data: { 
              labels: labels, 
              datasets:[{ 
                data: data, 
                backgroundColor: bgColors, 
                borderColor: '#1e1e1e', 
                borderWidth: 4,
                hoverOffset: 10
              }] 
            },
            options: { 
              responsive: true, 
              maintainAspectRatio: false, 
              cutout: '75%',
              plugins: { 
                legend: { position: 'bottom', labels: { color: 'white', padding: 20, font: {family: 'Inter'} } } 
              } 
            }
          });
        }
      } // End of return
    } // End of app()
